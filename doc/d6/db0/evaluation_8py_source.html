33def cross_validation(classifier, X, y, n_folds = 5):
34 """
35 Performs cross-validation and returns various performance metrics.
36
37 Parameters:
38 classifier (object): The classifier to be used.
39 X (DataFrame): Feature set.
40 y (Series): Label set.
41 n_folds (int): Number of folds for cross-validation. Default is 5.
42
43 Returns:
44 metrics_results (dict): Dictionary containing various performance metrics.
45 """
46 y_pred = cross_val_predict(estimator = classifier, X = X, y = y, cv = n_folds)
47
48 metrics_results = dict()
49 tn, fp, fn, tp = confusion_matrix(y, y_pred).ravel()
50 metrics_results['tn'] = tn
51 metrics_results['fp'] = fp
52 metrics_results['fn'] = fn
53 metrics_results['tp'] = tp
54 metrics_results['accuracy'] = skmetrics.accuracy_score(y, y_pred)
55 metrics_results['precision'] = skmetrics.precision_score(y, y_pred, zero_division = 0)
56 metrics_results['recall'] = skmetrics.recall_score(y, y_pred, zero_division = 0)
57 metrics_results['f1'] = skmetrics.f1_score(y, y_pred, zero_division = 0)
58 metrics_results['roc_auc'] = skmetrics.roc_auc_score(y, y_pred)
59 metrics_results['mcc'] = skmetrics.matthews_corrcoef(y, y_pred)
60 return metrics_results
