17def float_range(mini,maxi):
18 """
19 Creates a function to check if a float argument is within a specific range.
20
21 Parameters:
22 mini (float): Minimum value of the range (exclusive).
23 maxi (float): Maximum value of the range (exclusive).
24
25 Returns:
26 function: A function that checks if a float is within the given range.
27 """
28 def float_range_checker(arg):
29 try:
30 f = float(arg)
31 except ValueError:
32 raise argparse.ArgumentTypeError("Must be a Floating Point Number")
33 if f <= mini or f >= maxi:
34 raise argparse.ArgumentTypeError("Must be > " + str(mini) + " and < " + str(maxi))
35 return f
36 return float_range_checker

210def parallelize_func(func, parameters, cores = cpu_count()):
211 """
212 Parallelizes the execution of a function across multiple cores.
213
214 Parameters:
215 func (function): The function to parallelize.
216 parameters (list): List of parameters for the function.
217 cores (int): Number of cores to use. Default is the number of CPU cores.
218
219 Returns:
220 list: List of results from the parallelized function.
221 """
222 with Pool(cores) as pool:
223 results = list(pool.imap(func, parameters))
224 return results
