46def graph_class(data_df, args, method, dataset, label_threshold = 2.5):
47 """
48 Generates a stacked bar graph to visualize classification results.
49
50 Parameters:
51 data_df (DataFrame): DataFrame containing the classification results.
52 args (Namespace): Command-line arguments.
53 method (str): Feature selection method.
54 dataset (str): Name of the dataset file.
55 label_threshold (float): Threshold for displaying labels on bars. Default is 2.5.
56 """
57 sns.set(style = 'white')
58 models_index = list(data_df['model'].str.upper())
59 classification_dict = dict()
60 classification_list = ['TP', 'FP', 'TN', 'FN']
61 for classification in classification_list:
62 classification_dict[classification] = list(data_df[classification.lower()])
63
64 df = pd.DataFrame(classification_dict, index=models_index)
65 stacked_data = df.apply(lambda x: x * 100 / sum(x), axis = 1)
66
67 plt.figure(figsize = (10, 5))
68 bottom = pd.Series([0] * len(stacked_data), index = stacked_data.index)
69 colors = sns.color_palette(n_colors = len(classification_list))#'tab10', len(classification_list))
70
71 for i, classification in enumerate(classification_list):
72 sns.barplot(x = stacked_data[classification], y = stacked_data.index, left = bottom,
73 color = colors[i], label = classification)
74 bottom += stacked_data[classification]
75
76 plt.xlabel('Values (%)')
77 plt.ylabel('Model')
78 plt.ylim(-1, len(models_index))
79 plt.legend(ncol = len(classification_list), loc = 'upper center')
80 plt.title(f'Classification to {method} With Dataset in {os.path.basename(dataset)}')
81
82 # Add labels only if they are above the threshold
83 bottom = pd.Series([0] * len(stacked_data), index = stacked_data.index)
84 for i, classification in enumerate(classification_list):
85 for j, value in enumerate(stacked_data[classification]):
86 if value > label_threshold:
87 plt.text(bottom[stacked_data.index[j]] + value / 2, j, f'{value:.2f}', ha = 'center', va = 'center',
88 color = 'black', weight = 'bold', rotation = 90)
89 bottom += stacked_data[classification]

95 """

109 angles = [n / float(num_vars) * 2 * pi for n in range(num_vars)]
110 angles += angles[:1]
111 fig, ax = plt.subplots(figsize = (10, 8), subplot_kw = dict(polar = True))
112 for i in range(num_models):
113 values = df.loc[i].drop('Model').values.flatten().tolist()
114 values += values[:1]
115 ax.plot(angles, values, linewidth = 2, linestyle = 'solid', label = df['Model'][i])
116 ax.fill(angles, values, alpha = 0.25)
117
118 # Add Tags
119 plt.xticks(angles[:-1], categories)
120 plt.legend(loc = 'upper right', bbox_to_anchor = (0.1, 0.1))
121 plt.title(f'Comparison to {method} With Dataset in {os.path.basename(dataset)}')
